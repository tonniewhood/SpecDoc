\subsection{System Verification}

\subitem{
    % Verification of all System requirements.
}

\subsubsection{Processing}
\subitem{
    % Verification of Processing requirements.
}
\begin{enumerate}
    \item Verify sufficient compute for all modules.
    \item Verify support for real-time scheduling and isolation.
    \item Verify fault tolerance and recovery.
\end{enumerate}

\subsubsection{Built in HW Capabilities}
\subitem{
    % Verification of Built in HW Capabilities requirements.
}
\begin{enumerate}
    \item Verify provision of required sensors and actuators.
    \item Verify hardware health monitoring.
    \item Verify support for hardware expansion where feasible.
\end{enumerate}

\subsubsection{HW Interfaces}
\subitem{
    % Verification of HW Interfaces requirements.
}
\begin{enumerate}
    \item Verify support for standard communication protocols.
    \item Verify robust electrical and logical connections.
    \item Verify diagnostics for interface health.
\end{enumerate}

\subsubsection{HAL}
\subitem{
    % Verification of HAL requirements.
}
\begin{enumerate}
    \item Verify exposure of consistent APIs for hardware access.
    \item Verify isolation of hardware changes from application logic.
    \item Verify support for safe fallback on hardware faults.
\end{enumerate}

\subsubsection{Power Management}
\subitem{
    % Verification of Power Management requirements.
}
\begin{enumerate}
    \item Verify monitoring and regulation of power supply.
    \item Verify support for low-power and standby modes.
    \item Verify protection against overcurrent and undervoltage.
\end{enumerate}

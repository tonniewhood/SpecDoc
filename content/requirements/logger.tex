        Record mission data and provide mechanisms for transmission or storage by collecting telemetry, writing to storage, and/or sending data via communications links.

    \subsubsection{Non-Functional Requirements}

        \textbf{Energy}
        \begin{enumerate}[(a)]
            \item Logging shall not exceed allocated power/CPU budgets.
        \end{enumerate}

        \textbf{Environments}
        \begin{enumerate}[(a)]
            \item Maintain logging integrity across operating conditions.
        \end{enumerate}

        \textbf{Safety}
        \begin{enumerate}[(a)]
            \item Logging should not interfere with real-time control operations.
        \end{enumerate}

        \textbf{Structure}
        \begin{enumerate}[(a)]
            \item Provide schemas and retention policies for stored data.
        \end{enumerate}

        \textbf{Standards and Regulations}
        \begin{enumerate}[(a)]
            \item Follow applicable data retention and privacy regulations.
        \end{enumerate}

    \subsubsection{Submodules}
    \subitem{
        The Logger is comprised of two submodules that handle data persistence and transmission
        for mission telemetry and diagnostic information.
    }

    \paragraph{Storage}
    \subitem{
        The Storage submodule is responsible for persisting telemetry and logs to onboard non-volatile memory,
        ensuring data is retained for later retrieval and analysis.
    }

        \subparagraph{Functional Requirements}
        \begin{enumerate}[(a)]
            \item Support retrieval by timestamp and event markers.
        \end{enumerate}

        \subparagraph{Inputs}
        \begin{enumerate}[(a)]
            \item Telemetry streams to be persisted, storage configuration as defined in XXX:location
        \end{enumerate}

        \subparagraph{Outputs}
        \begin{enumerate}[(a)]
            \item Stored log files or records retrievable by time/index as defined in XXX:location
        \end{enumerate}

        \subparagraph{Non-Functional Requirements}

            \textbf{Energy}
            \begin{enumerate}[(a)]
                \item Execute within allocated CPU and power budgets.
            \end{enumerate}

            \textbf{Environments}
            \begin{enumerate}[(a)]
                \item Maintain performance across expected environmental conditions.
            \end{enumerate}

            \textbf{Safety}
            \begin{enumerate}[(a)]
                \item Protect critical logs and prevent data loss on faults.
            \end{enumerate}

            \textbf{Structure}
            \begin{enumerate}[(a)]
                \item Modular design with clear interfaces.
            \end{enumerate}

            \textbf{Standards and Regulations}

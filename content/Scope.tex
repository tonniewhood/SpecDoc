\section{SCOPE}



\begin{enumerate}[label=(\alph*)]
	\item \textbf{\underline{General:}}{
		This document defines the design and verification requirements for the 
		Uncrewed Aerial Vehicle (UAV) Flight Control Module (FCM). The system provides 
		the core functionality required for stable, autonomous, and remote operation
		of a fixed-wing aircraft. It manages flight stabilization, navigation, 
		control, and mission execution through integrated sensing, actuation,
		and computation. The FCM shall support interfacing 
		with an external companion system for higher-level mission planning and 
		enable autonomous operation in communication-limited environments, 
		including navigation to a designated area and precision landing.
		}
	\item \textbf{\underline{Acronyms:}}\begin{enumerate}[label=(\roman*)]
		\item UAV: Uncrewed Aerial Vehicle
		\item FCM: Flight Control Module
		\item HAL: Hardware Abstraction Layer
		\item GPS: Global Positioning System
		\item IMU: Inertial Measurement Unit
		\item PVA: Position/Velocity/Acceleration
		\item ATL: Attitude/Throttle-Level
		\item SD: Secure Digital
		% \item PWM: Pulse Width Modulation
		% \item API: Application Programming Interface
		\end{enumerate}
\end{enumerate}
